자주사용하는 입력어
ubuntu@sytrader.duckdns.org
cd ~/kiwoom-server
pm2 flush
pm2 logs kiwwm-server --lines 50
pm2 logs kiwwm-server --lines 100
node -c auto-trader-core.js
node -c server.js
node -c open-strategy.js
pm2 restart kiwwm-server --update-env
pm2 restart sy-quant-core --update-env

    리셋확인   :   curl http://localhost:3000/api/paper-state

------------------------------------------------------------------------------------------------------------------
장종료 후 분석화일 보여주기   →    복사해서 분석하면 됨

jq --arg today "$(date +%F)" '
.tradeLogs[]
| select((.date // "") | contains($today))
| {
    time,
    name,
    type,
    strategyGroup,
    strategyName,
    discoverScore,
    buyPrice,
    sellPrice,
    profit,
    profitRate,
    changeRate,
    volume
  }
' ~/kiwoom-server/paper-state.json
